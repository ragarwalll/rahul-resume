% modifiers/sections.tex - This file contains the section formatting for the XeLaTeX template
% Format the section titles
\titleformat{\section}%                  % Formatting \section command
{%                                       % Format of the section title
  \color{headings}%                    % Color we defined earlier (gray)
  \scshape%                            % Small caps
  \ifx\sectionfontpath\empty%
    \fontspec{\sectionfont}%
  \else%
    \fontspec[Path = \sectionfontpath]{\sectionfont}%
  \fi%
  \fontsize{\largefontsize}{\largefontheight}\selectfont%    % 16pt font size, 24pt line height
  \raggedright%                      % Align left
  \uppercase%                        % Make text uppercase
}
{} % Label format (empty here - no numbers)
{0em} % Separation between label and title
{} % Before-code (empty)

\renewcommand\section[1]{%
  \par%
  {\color{headings}%
    \scshape%
    \ifx\sectionfontpath\empty%
      \fontspec{\sectionfont}%
    \else%
      \fontspec[Path = \sectionfontpath]{\sectionfont}%
    \fi%
    \fontsize{\largefontsize}{\largefontheight}\selectfont%
    \MakeUppercase{#1}}\par%  % Changed \uppercase to \MakeUppercase and grouped
  \normalfont%
  \vspace{0.5em}%
 }
