\newcommand{\info}[1]{
  \color{subheadings}
  \raggedright
  \scshape
  \ifx\infofontpath\empty
    \fontspec{\infofont}
  \else
    \fontspec[Path = \infofontpath]{\infofont}
  \fi
  \fontsize{\infofontsize}{\infofontheight}
  \selectfont {#1 \\}
  \normalfont
}
