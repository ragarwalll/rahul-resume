\newcommand{\lastupdated}{\begin{textblock}{60} (155,0)
    % use the gray color that we defined earlier
    \color{date}
    \ifx\lastupdatedfontpath\empty
      \fontspec{\lastupdatedfont}
    \else
      \fontspec[Path = \lastupdatedfontpath]{\lastupdatedfont}
    \fi
    \fontsize{8pt}{10pt}
    \selectfont
    % set the date to today
    Last Updated on
    \today
    % end the text block
  \end{textblock}
}
